# Codex Auth Upstream Snapshot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 让 Codex 新增账号/登录链路回切到 upstream snapshot 驱动模式，使 `fetch`、`chat.headers`、`auth methods` 都与上游 `packages/opencode/src/plugin/codex.ts` 对齐，并移除本地自定义 `src/codex-oauth.ts` 登录实现。

**Architecture:** 先升级 `scripts/sync-codex-upstream.mjs` 与 `src/upstream/codex-plugin.snapshot.ts`，让 snapshot 能承载最新 upstream `codex.ts` 的完整 auth methods；再在 `src/upstream/codex-loader-adapter.ts` 暴露官方 auth methods 加载接口；最后把 `src/providers/codex-menu-adapter.ts` 的新增账号流程改成“选择官方 method、执行官方 authorize/callback、做最小结果归一化并落库”，并删掉 `src/codex-oauth.ts` 及其测试。整个实现坚持 TDD：先写失败测试，再做最小改动通过。

**Tech Stack:** TypeScript、Node 内置 `node:test`、现有 upstream snapshot sync 脚本、现有 Codex store/menu adapter/plugin hooks 结构。

function buildShimBlock() {
  return `/* LOCAL_SHIMS_START */
type RequestInfo = Request | URL | string

type Hooks = any
type PluginInput = any
const officialCodexExportBridgeStorage = new AsyncLocalStorage<{
  fetchImpl: typeof globalThis.fetch
  version: string
}>()

const officialCodexExportBridge = {
  version: "snapshot",
  fetchImpl(request, init) {
    return globalThis.fetch(request, init)
  },
  async run(options = {}, fn) {
    return officialCodexExportBridgeStorage.run(
      {
        fetchImpl: options.fetchImpl ?? this.fetchImpl,
        version: options.version ?? this.version,
      },
      fn,
    )
  },
}

const Installation = {
  get VERSION() {
    return officialCodexExportBridgeStorage.getStore()?.version ?? officialCodexExportBridge.version
  },
  set VERSION(value) {
    officialCodexExportBridge.version = value
  },
}

const OAUTH_DUMMY_KEY = "official-codex-oauth"

function fetch(request: RequestInfo | URL, init?: RequestInit) {
  return (officialCodexExportBridgeStorage.getStore()?.fetchImpl ?? officialCodexExportBridge.fetchImpl)(request, init)
}

function sleep(ms) {
  return Bun.sleep(ms)
}

const Bun = {
  serve(options) {
    const server = createServer((req, res) => {
      const request = new Request(`http://127.0.0.1:${options.port}${req.url ?? "/"}`, {
        method: req.method,
        headers: req.headers,
      })
      Promise.resolve(options.fetch(request)).then(async (response) => {
        res.statusCode = response.status
        response.headers.forEach((value, key) => res.setHeader(key, value))
        res.end(await response.text())
      }).catch((error) => {
        res.statusCode = 500
        res.end(String(error))
      })
    })
    server.listen(options.port)
    return {
      stop() {
        server.close()
      },
    }
  },
  sleep(ms) {
    return new Promise((resolve) => setTimeout(resolve, ms))
  },
}

const Log = {
  create() {
    return {
      info() {},
      warn() {},
      error() {},
    }
  },
}
/* LOCAL_SHIMS_END */`
}

ensureSingleMatch(normalized, /methods\s*:\s*\[/g, "auth.methods")

type OfficialCodexAuthMethod = Awaited<ReturnType<typeof loadOfficialCodexAuthMethods>>[number]

type AdapterDependencies = {
  client: AuthClient
  now?: () => number
  loadOfficialCodexAuthMethods?: typeof loadOfficialCodexAuthMethods
  selectOfficialCodexAuthMethod?: (methods: OfficialCodexAuthMethod[]) => Promise<OfficialCodexAuthMethod | undefined>

async function selectOfficialCodexAuthMethodDefault(methods: OfficialCodexAuthMethod[], prompt: (message: string) => Promise<string>) {
  const browser = methods.find((method) => method.type === "oauth" && /browser/i.test(method.label))
  const headless = methods.find((method) => method.type === "oauth" && /headless/i.test(method.label))
  const value = (await prompt("OpenAI/Codex login mode ([1] browser, [2] headless, Enter to cancel): ")).trim().toLowerCase()
  if (!value) return undefined
  if (value === "1" || value === "browser" || value === "b") return browser
  if (value === "2" || value === "headless" || value === "h" || value === "device") return headless
  return undefined
}

const loadOfficialMethods = inputDeps.loadOfficialCodexAuthMethods ?? loadOfficialCodexAuthMethods
const selectOfficialMethod = inputDeps.selectOfficialCodexAuthMethod ?? ((methods) => selectOfficialCodexAuthMethodDefault(methods, prompt))

authorizeNewAccount: async () => {
  const methods = (await loadOfficialMethods({ client: inputDeps.client }))
    .filter((method) => method.type === "oauth")
  const selected = await selectOfficialMethod(methods)
  if (!selected?.authorize) return undefined

  const authorization = await selected.authorize()
  if (!authorization) return undefined
  if (authorization.method !== "auto") {
    throw new Error(`Unsupported official Codex auth method: ${authorization.method}`)

  const result = await authorization.callback()

  if (!result || result.type !== "success" || !("refresh" in result)) return undefined

  await inputDeps.client.auth.set({
    path: { id: "openai" },

      type: "oauth",
      refresh: result.refresh,
      access: result.access,

    },

    refresh: result.refresh,
    access: result.access,
